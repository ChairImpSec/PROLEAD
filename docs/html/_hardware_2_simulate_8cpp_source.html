4uint64_t Hardware::Simulate::EvaluateOperation(Hardware::OperationStruct Operation, uint64_t *Values, char Intermediate)
5{
6 uint64_t Res = 0;
7 int i, j;
8
9 for (i = 0; i < Operation.NumberOfClauses; i++)
10 {
11 Res = Values[Operation.OperandsInClause[i][0]];
12
13 if (Operation.OperationOfClause[i] == Operation_AND)
14 for (j = 1; j < Operation.NumberOfOperandsInClause[i]; j++)
15 Res &= Values[Operation.OperandsInClause[i][j]];
16 else if (Operation.OperationOfClause[i] == Operation_OR)
17 for (j = 1; j < Operation.NumberOfOperandsInClause[i]; j++)
18 Res |= Values[Operation.OperandsInClause[i][j]];
19 else if (Operation.OperationOfClause[i] == Operation_XOR)
20 for (j = 1; j < Operation.NumberOfOperandsInClause[i]; j++)
21 Res ^= Values[Operation.OperandsInClause[i][j]];
22 else
23 Res = ~Res;
24
25 Values[Intermediate + i] = Res;
26 }
27
28 return Res;
29}
