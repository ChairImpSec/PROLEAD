593
594
595
596
597 if ((Settings.TestMultivariate != 0) && (Settings.NumberOfTestClockCycles > 1) && (Settings.TestOrder != 1)){
598 // Multivariate setting
599 Test.ProbingSet.at(SimulationIndex).resize(Software::Operators::BinomialCoefficient(NumberOfStandardProbes,Settings.TestOrder), Settings.TestOrder);
600 // Set the bitmask to the first possible probe combination

621

623 }else{
624 // Univariate setting
625 // Store standard probes for each clock cycle separately
626 CycleIndex = 0;
627
628 //as StandardProbesPerSimulation has ascending ClockCycle -> if we found a ClockCycle that is greater we know all following Probes are also in later ClockCycle
629 if(Settings.TestOrder == 1){

645

648 NumberOfStandardProbesPerCycle = StandardProbeIndex - CycleIndex;
649 //make space for new pairs
650 Test.ProbingSet.at(SimulationIndex).resize(Test.ProbingSet.at(SimulationIndex).size() + Software::Operators::BinomialCoefficient(NumberOfStandardProbesPerCycle,Settings.TestOrder), (uint32_t)Settings.TestOrder);

661

663 if (Test.CombinationBitmask.at(SetIndex)){
664 Test.Combination.at(ProbeIndex) = ThreadSimulation.StandardProbesPerSimulation.at(SimulationIndex).at(SetIndex + CycleIndex);
665 ProbeIndex++;
666 }
667 if((ProbeIndex >= Settings.TestOrder)){
668 break;
669 }
670 }

673 ProbingSetIndex++;
674
675 } while (std::prev_permutation(Test.CombinationBitmask.begin(), Test.CombinationBitmask.end()));

678 }
679 }
680
681 //remove complete redundant ProbingSets
682 //sort and unique
683 std::sort(Test.ProbingSet.at(SimulationIndex).begin(), Test.ProbingSet.at(SimulationIndex).end(), ProbingSetCompare);

686 }
