180void Software::Probing::CreatePipelineForwardingProbe(std::vector<Software::ProbesStruct>& StandardProbes, std::vector<uint8_t>& PipelineForwardingProbes, uint8_t BitIdx, uint32_t& ProbeIndex, uint64_t ProbeInfo, uint32_t NrOfPipelineStages, std::vector<::mulator::CPU_State>& pipeline_cpu_states){
181
182
183 uint32_t bit_pos_ctr = 0;
184
185 //use TransitionCycles and SpecialInfo to store the probe values. both together have space for 128 bits. We can have at max. a 6 stage pipeline. If all 17 registers (r0-r16) are probed in 6 stages we get: 17*6 = 102
186 uint64_t tmp_TransitionCycles = 0;
187 uint64_t tmp_SpecialInfo = 0;
188
189 for(uint32_t pipeline_idx = 0; pipeline_idx < NrOfPipelineStages; ++pipeline_idx){ //go over every pipeline stage
190 if(pipeline_cpu_states.at(pipeline_idx).containing_valid_pipeline_values){ //only add values if registers are valid
191 for(const auto& RegIdx : PipelineForwardingProbes){
192 if(bit_pos_ctr >= 64){
193 tmp_SpecialInfo |= (static_cast<uint64_t>((((pipeline_cpu_states.at(pipeline_idx).registers[RegIdx] >> BitIdx) & 0x1)) << (bit_pos_ctr - 64)));
194 }
195 else{
196 tmp_TransitionCycles |= (static_cast<uint64_t>((((pipeline_cpu_states.at(pipeline_idx).registers[RegIdx] >> BitIdx) & 0x1)) << (bit_pos_ctr)));
197 }
198 bit_pos_ctr++;
199 }
200 }
201 }
202
203 StandardProbes.at(ProbeIndex).TransitionCycles = tmp_TransitionCycles;
204 StandardProbes.at(ProbeIndex).SpecialInfo = tmp_SpecialInfo;
205
206
207 StandardProbes.at(ProbeIndex).ProbeInfo = (ProbeInfo | (BitIdx << BIT_OFFSET) | (bit_pos_ctr));
208 ProbeIndex++;
