36void ProbeExtension<RobustProbeExtension>::PropagateProbe(
37 LibraryStruct& library, CircuitStruct& circuit, SettingsStruct& settings) {
38 std::vector<bool> probe_already_considered(circuit.NumberOfSignals, false);
39 std::queue<int> signals_in_progress;
40 int current_signal_index, input_index, next_signal_index;
41
42 signals_in_progress.push(signal_index_);
43
44 while (!signals_in_progress.empty()) {
45 current_signal_index = signals_in_progress.front();
46 signals_in_progress.pop();
47
48 if (!(circuit.Signals[current_signal_index]->Deleted) &&
49 (current_signal_index != settings.ClockSignal) &&
50 !probe_already_considered[current_signal_index]) {
51 if ((circuit.Signals[current_signal_index]->Output == -1) ||
52 (library
53 .CellTypes
54 [circuit
55 .Cells[circuit.Signals[current_signal_index]->Output]
56 ->Type]
57 ->GateOrReg == CellType_Reg)) {
58 if ((circuit.Signals[current_signal_index]->ProbeAllowed) ||
59 (circuit.Signals[signal_index_]->ProbeAllowed == 2)) {
60 extension_indices_.push_back(current_signal_index);
61 }
62 } else {
63 if (!circuit.Cells[circuit.Signals[current_signal_index]->Output]
64 ->Deleted) {
65 for (input_index = 0;
66 input_index <
67 circuit.Cells[circuit.Signals[current_signal_index]->Output]
68 ->NumberOfInputs;
69 ++input_index) {
70 next_signal_index =
71 circuit.Cells[circuit.Signals[current_signal_index]->Output]
72 ->Inputs[input_index];
73
74 if (!probe_already_considered[next_signal_index]) {
75 signals_in_progress.push(next_signal_index);
76 }
77 }
78 }
79 }
80 }
81
82 probe_already_considered[current_signal_index] = true;
83 }
84
85 std::sort(extension_indices_.begin(), extension_indices_.end());
86 extension_indices_.erase(
87 std::unique(extension_indices_.begin(), extension_indices_.end()),
88 extension_indices_.end());

312void ProbingSet<RobustProbeExtension>::NormalTableUpdate(
313 SimulationStruct& simulation, std::vector<Probe>& extended_probes) {
314 unsigned int group_index;
315 unsigned char hash_value0, hash_value1;
316 hardware::stats::Key key;
317 size_t bit_index, key_index, tmp_index;
318 size_t number_of_layers = contingency_table_.GetNumberOfLayers();
319 size_t number_of_extended_probes = GetNumberOfProbeExtensions();
320 size_t key_size = contingency_table_.GetKeySizeExcludingHashValues();
321 std::vector<unsigned int> probe_extension_indices = GetAllProbeExtensions();
322 std::vector<unsigned int> signal_indices(number_of_extended_probes);
323 std::vector<unsigned int> cycles(number_of_extended_probes);
324
325 for (bit_index = 0; bit_index < number_of_extended_probes; ++bit_index) {
326 signal_indices[bit_index] =
327 simulation.GlitchExtendedProbeIndex_of_Signal
328 [extended_probes[probe_extension_indices[bit_index]]
329 .GetSignalIndex()];
330 cycles[bit_index] =
331 extended_probes[probe_extension_indices[bit_index]].GetCycle();
332 }
333
334 for (uint64_t sim_index = 0; sim_index < simulation.NumberOfStepSimulations;
335 ++sim_index) {
336 hash_value0 = 0;
337 hash_value1 = 0;
338 key = std::make_unique<unsigned char[]>(key_size);
339
340 // Set first hash value
341 for (bit_index = 0;
342 (bit_index < 8) && (bit_index < number_of_extended_probes);
343 ++bit_index) {
344 hash_value0 <<= 1;
345 hash_value0 |= simulation.ProbeValues[sim_index][cycles[bit_index]]
346 [signal_indices[bit_index]];
347 }
348
349 // Set second hash value
350 for (bit_index = 8;
351 (bit_index < 16) && (bit_index < number_of_extended_probes);
352 ++bit_index) {
353 hash_value1 <<= 1;
354 hash_value1 |= simulation.ProbeValues[sim_index][cycles[bit_index]]
355 [signal_indices[bit_index]];
356 }
357
358 // Get regular key
359 for (key_index = 0; key_index < key_size; ++key_index) {
360 tmp_index = key_size - key_index - 1;
361 for (bit_index = (key_index + number_of_layers) << 3;
362 bit_index < ((key_index + number_of_layers + 1) << 3) &&
363 (bit_index < number_of_extended_probes);
364 ++bit_index) {
365 key[tmp_index] <<= 1;
366 key[tmp_index] |= simulation.ProbeValues[sim_index][cycles[bit_index]]
367 [signal_indices[bit_index]];
368 }
369 }
370
371 group_index = simulation.SelectedGroups[sim_index];
372 contingency_table_.UpdateTable(std::move(key), hash_value0, hash_value1,
373 simulation.NumberOfGroups, group_index);
374 }

448 std::vector<unsigned int>& standard_probe_indices,

450 unsigned int standard_probe_index;
451 std::vector<unsigned int> cycles(standard_probe_indices.size());

468 unsigned int signal_index, unsigned int clock_cycle) {
469 Probe probe(signal_index, clock_cycle);
470 std::vector<Probe>::iterator it = std::lower_bound(
471 extended_probes_.begin(), extended_probes_.end(), probe,
472 [](const Probe& lhs, const Probe& rhs) { return lhs < rhs; });
473 std::iterator_traits<std::vector<Probe>::iterator>::difference_type diff =
474 std::distance(extended_probes_.begin(), it);
475
476 if ((it == extended_probes_.end()) || (extended_probes_.at(diff) != probe)) {
477 throw std::runtime_error(
478 "A glitch-extended probe was not found in the list of glitch-extended "
479 "probes.");
480 }
481
482 return (size_t)diff;

487 std::vector<unsigned int>& standard_probe_indices,
488 bool is_with_transitional_leakage, SettingsStruct& settings) {
489 size_t probe_index, signal_index;
490 unsigned int probe_extension_index, clock_cycle;
491 std::vector<unsigned int> signal_indices;

1153template <>
1154void Adversaries<RobustProbeExtension>::EvaluateProbingSets(
1155 LibraryStruct& library, CircuitStruct& circuit, SettingsStruct& settings,
1156 SharedDataStruct* shared_data, SimulationStruct& simulation,
1157 timespec& start_time, unsigned int& probe_step_index) {
1158 std::vector<double> number_of_simulations_per_group(settings.NumberOfGroups,
1159 0.0);
1160 std::vector<boost::mt19937> thread_rng(settings.Max_no_of_Threads);
1161 uint64_t step_simulation_index, simulation_index;
1162 omp_set_num_threads(settings.Max_no_of_Threads);
1163 double elapsed_time_period, maximum_g_value_deleted = 0.0;
1164 int thread_index;
1165 std::string printed_probing_set_deleted;
1166 size_t space, table_width;
1167 bool is_in_compact_mode = settings.CompactDistributions == 1;
1168
1169 std::cout << "Evaluate security under the robust probing model!" << std::endl;
1170
1171 for (size_t set_index = 0; set_index < GetNumberOfProbingSets();
1172 ++set_index) {
1173 probing_sets_[set_index].Initialize(settings.NumberOfGroups,
1174 is_in_compact_mode);
1175 }
1176
1177 SetUniqueProbes();
1178
1179 // We use a PRNG that is thread safe
1180 Util::GenerateThreadRng(thread_rng, settings.Max_no_of_Threads);
1181 space = PrintEvaluationHeader(circuit);
1182 table_width = space + 89;
1183
1184 for (step_simulation_index = 0;
1185 step_simulation_index <
1186 (settings.NumberOfSimulations / settings.NumberOfStepSimulations);
1187 ++step_simulation_index) {
1188#pragma omp parallel for schedule(guided) private(thread_index)
1189 for (simulation_index = 0;
1190 simulation_index < (settings.NumberOfStepSimulations / 64);
1191 ++simulation_index) {
1192 thread_index = omp_get_thread_num();
1193 Hardware::Simulate::All(library, circuit, settings,
1194 &shared_data[thread_index], simulation,
1195 simulation_index, thread_rng[thread_index]);
1196 }
1197
1198 simulation.NumberOfProcessedSimulations =
1199 (step_simulation_index + 1) * settings.NumberOfStepSimulations;
1200
1201 for (simulation_index = 0;
1202 simulation_index < simulation.NumberOfStepSimulations;
1203 ++simulation_index) {
1204 number_of_simulations_per_group[simulation
1205 .SelectedGroups[simulation_index]] +=
1206 1.0;
1207 }
1208
1209 Test(settings, simulation, number_of_simulations_per_group,
1210 is_in_compact_mode);
1211
1212 elapsed_time_period = Util::EndClock(start_time);
1213 PrintEvaluationBody(circuit, settings, simulation, maximum_g_value_deleted,
1214 printed_probing_set_deleted, elapsed_time_period,
1215 space);
1216
1217 if ((simulation.NumberOfProcessedSimulations %
1218 settings.NumberOfStepSimulationsToWrite) == 0) {
1219 PrintReport(circuit, settings, simulation, probe_step_index, table_width);
1220 }
1221
1222 if (settings.RemoveProbingSets &&
1223 !settings.CompactDistributions) { // Remove probing set if enough
1224 // traces are processed
1225 RemoveProbingSetsWithEnoughTraces(
1226 circuit, simulation.NumberOfProcessedSimulations,
1227 maximum_g_value_deleted, printed_probing_set_deleted);
1228
1229 if (GetNumberOfProbingSets() == 0) {
1230 if ((simulation.NumberOfProcessedSimulations %
1231 settings.NumberOfStepSimulationsToWrite) != 0) {
1232 PrintReport(circuit, settings, simulation, probe_step_index,
1233 table_width);
1234 }
1235 break;
1236 }
1237 }
1238 }
1239}

1281 }

1287 PrintProbingSetInformation();
