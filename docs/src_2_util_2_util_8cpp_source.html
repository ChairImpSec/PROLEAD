346template<class BucketContainer>
347void ContingencyTable<BucketContainer>::ComputeNumberOfRequiredTraces(size_t number_of_groups, double beta_threshold, double effect_size){
348 unsigned int factor = 1;
349 size_t size_of_table = GetNumberOfEntries();
350 double non_centrality_parameter, beta;
351 bool too_small = true;
352
353 number_of_required_traces_ = 1;
354
355 if (size_of_table > 1) {
356 size_t degree_of_freedom = (number_of_groups - 1) * (size_of_table - 1);
357
358 // Compute critical g value
359 boost::math::chi_squared_distribution<> distribution1(degree_of_freedom);
360 double critical_value = boost::math::quantile(
361 boost::math::complement(distribution1, beta_threshold));
362
363 while (true) {
364 non_centrality_parameter = effect_size * effect_size * number_of_required_traces_;
365
366 // Get type II error rate
367 boost::math::non_central_chi_squared_distribution<> distribution2(
368 degree_of_freedom, non_centrality_parameter);
369 beta = boost::math::cdf(distribution2, critical_value);
370
371 if (beta > beta_threshold) {
372 if (too_small) {
373 number_of_required_traces_ += factor;
374 factor <<= 1;
375 } else {
376 if (factor == 1) {
377 break;
378 } else {
379 too_small = true;
380 factor = 1;
381 number_of_required_traces_ += factor;
382 }
383 }
384 } else {
385 if (!too_small) {
386 number_of_required_traces_ -= factor;
387 factor <<= 1;
388 } else {
389 if (factor == 1) {
390 break;
391 } else {
392 too_small = false;
393 factor = 1;
394 number_of_required_traces_ -= factor;
395 }
396 }
397 }
398 }
399 }

434void Util::ContingencyTableStruct::CalculateTraces(int NumberOfGroups, double BetaThreshold, double EffectSize){
435 bool TooSmall = true;
436 size_t BitIndex, GroupIndex;
437 unsigned int Factor = 1;
438 unsigned int TableSize = Entries.size();
439 double NoncentralityParameter, Beta;
440
441 Traces = 1;
442
443 for (GroupIndex = 0; GroupIndex < OnlyOneEntry.size(); GroupIndex++){
444 for (BitIndex = 0; BitIndex < OnlyOneEntry.at(GroupIndex).size(); BitIndex++){
445 TableSize += OnlyOneEntry.at(GroupIndex).at(BitIndex).size();
446 }
447 }
448
449 if (TableSize > 1){
450 // Compute degree of freedom
451 unsigned int Df = (NumberOfGroups - 1) * (TableSize - 1);
452
453 // Compute critical g value
454 boost::math::chi_squared_distribution<> Distribution1(Df);
455 double CriticalValue = boost::math::quantile(boost::math::complement(Distribution1, BetaThreshold));
456

458 // Get non-centrality parameter
459 NoncentralityParameter = EffectSize * EffectSize * Traces;
460
461 // Get type II error rate

464
465 if (Beta > BetaThreshold){
466 if (TooSmall){
467 Traces += Factor;
468 Factor <<= 1;
469 }else{
470 if (Factor == 1){
471 break;
472 }else{
473 TooSmall = true;
474 Factor = 1;

477 }
478 }else{

496double Util::GetGValue(unsigned int NumberOfGroups, unsigned int NumberOfSimulations, Util::ContingencyTableStruct& Table, std::vector<double> Frequency, unsigned int TableSize, unsigned int& Df){
497 unsigned int SumOverBin = 0, BinIndex = 0, BitIndex = 0, SanitisedTableSize = 0;
498 unsigned int GroupIndex = 0, GroupIndex2 = 0;
499 double gValue = 0.0, portion = 0.0, product = 0.0;
500 std::vector<double> ExpectedFrequency(NumberOfGroups, 0.0);
501 std::vector<unsigned int> PooledCount(NumberOfGroups, 0);
502 bool Valid = true;
503 Df = 0;
504
505 double PoolingFactor = (double)NumberOfSimulations / TableSize;
506 std::transform(Frequency.begin(), Frequency.end(), Frequency.begin(), [NumberOfSimulations](double x){return x / NumberOfSimulations;});
507
508 for (BinIndex = 0; BinIndex < Table.Entries.size(); BinIndex++){
509 SumOverBin = std::accumulate(Table.Entries.at(BinIndex).Count.begin(), Table.Entries.at(BinIndex).Count.end(), 0);
510
511 Valid = true;
512 for (GroupIndex = 0; GroupIndex < NumberOfGroups; GroupIndex++){
513 ExpectedFrequency.at(GroupIndex) = Frequency.at(GroupIndex) * SumOverBin;
514
515 if ((ExpectedFrequency.at(GroupIndex) < 5.0) || ((ExpectedFrequency.at(GroupIndex) < 5.0 * PoolingFactor) && ExpectedFrequency.at(GroupIndex) < 20.0)){
516 Valid = false;
517 }
518 }
519
520 if (Valid){
521 for (GroupIndex = 0; GroupIndex < NumberOfGroups; GroupIndex++){
522 if (Table.Entries.at(BinIndex).Count.at(GroupIndex)){
523 portion = Table.Entries.at(BinIndex).Count.at(GroupIndex) / ExpectedFrequency.at(GroupIndex);
524 product = Table.Entries.at(BinIndex).Count.at(GroupIndex) * std::log(portion);
525 gValue += product;
526 }
527 }
528
529 SanitisedTableSize++;
530 }else{
531 for (GroupIndex = 0; GroupIndex < NumberOfGroups; GroupIndex++){
532 PooledCount.at(GroupIndex) += Table.Entries.at(BinIndex).Count.at(GroupIndex);
533 }
534 }
535 }
536
537 for (GroupIndex = 0; GroupIndex < (unsigned int)Table.OnlyOneEntry.size(); GroupIndex++){
538 for (BitIndex = 0; BitIndex < Table.OnlyOneEntry.at(GroupIndex).size(); BitIndex++){
539 for (BinIndex = 0; BinIndex < Table.OnlyOneEntry.at(GroupIndex).at(BitIndex).size(); BinIndex++){
540 SumOverBin = 1;
541
542 Valid = true;
543 for (GroupIndex2 = 0; GroupIndex2 < NumberOfGroups; GroupIndex2++){
544 ExpectedFrequency.at(GroupIndex2) = Frequency.at(GroupIndex2) * SumOverBin;
545
546 if ((ExpectedFrequency.at(GroupIndex2) < 5.0) || ((ExpectedFrequency.at(GroupIndex2) < 5.0 * PoolingFactor) && ExpectedFrequency.at(GroupIndex2) < 20.0)){
547 Valid = false;
548 }
549 }
550
551 if (Valid){
552 for (GroupIndex2 = 0; GroupIndex2 < NumberOfGroups; GroupIndex2++){
553 portion = 1.0 / ExpectedFrequency.at(GroupIndex);
554 product = std::log(portion);
555 gValue += product;
556 }
557
558 SanitisedTableSize++;
559 }else{
560 PooledCount.at(GroupIndex) += 1;
561 }
562 }
563 }
564 }
565
566 SumOverBin = std::accumulate(PooledCount.begin(), PooledCount.end(), 0);
567
568 if (SumOverBin){
569 for (GroupIndex = 0; GroupIndex < NumberOfGroups; GroupIndex++){
570 ExpectedFrequency.at(GroupIndex) = Frequency.at(GroupIndex) * SumOverBin;
571
572 if (PooledCount.at(GroupIndex)){
573 portion = PooledCount.at(GroupIndex) / ExpectedFrequency.at(GroupIndex);
574 product = PooledCount.at(GroupIndex) * std::log(portion);
575 gValue += product;
576 }
577 }
578
579 SanitisedTableSize++;
580 }
581
582 gValue *= 2;
583 Df = (NumberOfGroups - 1) * (SanitisedTableSize - 1);
584
585 if (gValue < 0.0){
586 gValue = 0.0;
587 }
588
589 return gValue;
